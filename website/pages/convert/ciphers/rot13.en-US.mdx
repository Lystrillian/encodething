import { useState } from 'react';
import Textarea from '@textarea';
import { CustomLink } from '@link';
import useRot13 from '@hooks/ciphers/useRot13';

export default function ROT13Converter() {
  const [inputText, setInputText] = useState('');
  const { encodedValue, decodedValue } = useRot13(inputText);

  return (
    <div>
      <h2>ROT13 Converter</h2>
      <Textarea
        value={inputText}
        onChange={(e) => setInputText(e.target.value)}
        placeholder="Enter text to convert to ROT13"
      />
      <div>
        <h3>Encoded Value:</h3>
        <p>{encodedValue}</p>
      </div>
      <div>
        <h3>Decoded Value:</h3>
        <p>{decodedValue}</p>
      </div>
      <div>
        <h3>Documentation:</h3>
        <CustomLink href="https://en.wikipedia.org/wiki/ROT13">
          ROT13 on Wikipedia
        </CustomLink>
      </div>
    </div>
  );
}
