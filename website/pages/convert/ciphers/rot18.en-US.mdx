import { useState } from 'react';
import Textarea from '@textarea';
import { CustomLink } from '@link';
import useRot18 from '@hooks/ciphers/useRot18';

export default function ROT18Converter() {
  const [inputText, setInputText] = useState('');
  const { encodedValue, decodedValue } = useRot18(inputText);

  return (
    <div>
      <h2>ROT18 Converter</h2>
      <Textarea
        value={inputText}
        onChange={(e) => setInputText(e.target.value)}
        placeholder="Enter text to convert to ROT18"
      />
      <div>
        <h3>Encoded Value:</h3>
        <p>{encodedValue}</p>
      </div>
      <div>
        <h3>Decoded Value:</h3>
        <p>{decodedValue}</p>
      </div>
      <div>
        <h3>Documentation:</h3>
        <CustomLink href="https://en.wikipedia.org/wiki/ROT18">
          ROT18 on Wikipedia
        </CustomLink>
      </div>
    </div>
  );
}
